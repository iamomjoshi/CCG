                4.write a cpp program to find maximum out of two numbers using friend function.

                #include<iostream>
                using namespace std;

                class b; // Forward declaration

                class a{
                    int n;
                public:
                    a(int x){
                        n = x;
                    }
                    friend void greatest(a a1, b b1);
                };

                class b{
                    int n1;
                public:
                    b(int x){
                        n1 = x;
                    }
                    friend void greatest(a a1, b b1);
                };

                void greatest(a a1, b b1){
                    if(a1.n > b1.n1){
                        cout<<"number in class a is greatest: "<<a1.n<<endl;
                    }
                    else if(a1.n < b1.n1){
                        cout<<"number in class b is greatest: "<<b1.n1<<endl;
                    }
                    else{
                        cout<<"both numbers are equal"<<endl;
                    }
                }

                int main(){
                    cout<<"Program to find greatest of 2 numbers in 2 different classes using friend function"<<endl;
                    int n1, n2;
                    cout<<"Enter number for class a: "<<endl;
                    cin>>n1;
                    a a1(n1);
                    cout<<"Enter number for class b: "<<endl;
                    cin>>n2;
                    b b1(n2);
                    greatest(a1, b1);
                    cout<<"/n";
                    return 0;

VIRTUAL FUNCTIONS


            #include<iostream>
            using namespace std;

            class base{
            public:
                void display(){
                    cout << "\ndisplay base" << endl;
                }
                virtual void show(){
                    cout << "\n show base" << endl;
                }
            };

            class derived : public base{
            public:
                void display(){
                    cout << "\ndisplay derived" << endl;
                }
                
                void show(){
                    cout << "\n show derived" << endl;
                }
            };

            int main(){
                base b;
                derived d;
                base *bptr;
                
                cout << "\n bptr points to base:\n" << endl;
                bptr = &b;
                bptr->display();
                bptr->show();
                
                cout << "\n\nbptr points to derived:\n" << endl;
                bptr = &d;
                bptr->display();
                bptr->show();
                
                return 0;
            }
