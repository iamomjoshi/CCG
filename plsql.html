                9.1 Create a table named student having columns id integer, name character variable width 10
                columns, mobile character variable width 10 columns and email character variable width 10
                columns. Create a sequence seq_stud starting from 101 and increment by 1. Create a procedure
                val_ins1 to insert the rows into student table. Call the procedure and enter 5 meaningful rows.
                
                create table student (id number(5),name varchar2(10), mobile varchar2(10),email
                varchar2(25));
                create sequence seq_stud
                start with 101
                increment by 1;
                create or replace procedure val_ins1(S_name IN varchar2,s_mobile IN varchar2, S_email IN
                varchar2) is
                begin
                insert into student values (seq_stud.nextval, S_name,s_mobile, S_email);
                commit;
                end val_ins1;

                9.2 Write a PL/SQL program to display the student name and mobile no of a student whose id is
                will be accepted from user, also handle an error if data is not found using system defined
                exception.
                
                DECLARE
                s_id student.id%type;
                s_name student.Name%type;
                s_mobile student.mobile%type;

                s_id:=&Student_ID;

                10.3 create a package s_package which should include the procedure to add student, delete
                the student and display the names of students.
                
                CREATE OR REPLACE PACKAGE s_package AS
                -- Adds a student
                PROCEDURE addStudent(s_id student.id%type,
                s_name student.Name%type,
                s_mobile student.mobile%type,
                s_email student.email%type);
                -- Removes a student
                PROCEDURE delStudent(s_id student.id%TYPE);
                --Lists all student
                PROCEDURE listStudent;
                END s_package;
                /
                
                CREATE OR REPLACE PACKAGE BODY s_package AS
                PROCEDURE addStudent(s_id student.id%type,
                s_name student.Name%type,
                s_mobile student.mobile%type,
                s_email student.email%type)
                IS
                BEGIN
                INSERT INTO student (id,name,mobile,email)
                VALUES(s_id, s_name, s_mobile, s_email);
                END addStudent;
                PROCEDURE delStudent(s_id student.id%type) IS
                BEGIN
                DELETE FROM student
                WHERE id = s_id;
                END delStudent;
                
                PROCEDURE liststudent IS
                CURSOR s_student is
                SELECT name FROM student;
                TYPE s_list is TABLE OF student.Name%type;
                name_list s_list := s_list();
                counter integer :=0;
                BEGIN
                FOR n IN s_student LOOP
                counter := counter +1;
                name_list.extend;
                name_list(counter) := n.name;
                dbms_output.put_line('Student (' ||counter|| ') '||name_list(counter));
                END LOOP;
                END liststudent;
                END s_package;
                /
